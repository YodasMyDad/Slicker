Add & Remove

Dynamic slides

1
var addRemove = Slicker.init('.add-remove', {
  slidesToShow: 3,
  slidesToScroll: 3
});

var slideIndex = 1;
document.querySelector('.js-add-slide').addEventListener('click', function() {
  slideIndex++;
  addRemove.slickAdd('
' + slideIndex + '
');
});

document.querySelector('.js-remove-slide').addEventListener('click', function() {
  addRemove.slickRemove(slideIndex - 1);
  if (slideIndex > 1) slideIndex--;
});

var filtering = Slicker.init('.filtering', {
  slidesToShow: 4,
  slidesToScroll: 4
});

var filtered = false;
document.querySelector('.js-filter').addEventListener('click', function() {
  if (!filtered) {
    filtering.slickFilter((_, el) => parseInt(el.textContent, 10) % 2 === 0);
    this.textContent = 'Unfilter Slides';
  } else {
    filtering.slickUnfilter();
    this.textContent = 'Filter Slides';
  }
  filtered = !filtered;
});
